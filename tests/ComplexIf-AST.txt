Class {
	name = "ComplexIf",
	fields = [],
	methods = [
		Method {
			retType = boolean,
			name = "f",
			params = [
				VarDecl {
					type = char,
					name = "a",
					val = None
				}
			],
			body = Block ([
				If {
					cond = Binary {
						op = Equal,
						left = LocalOrFieldVar("c"),
						right = Char('a')
					},
					body = Return(Bool(true)),
					else = Some (
						If {
							cond = Binary {
								op = Equal,
								left = LocalOrFieldVar("c"),
								right = Char('b')
							},
							body = Return(Bool(false)),
							else = Some (
								cond = Binary {
									op = Equal,
									left = LocalOrFieldVar("c"),
									right = Char('c')
								},
								body = Return(Bool(true)),
								else = Block([
									If {
										cond = Binary {
											op = Or,
											left = Binary {
												op = Equal,
												left = LocalOrFieldVar("c"),
												right = Char('d')
											},
											right = Binary {
												op = Equal,
												left = LocalOrFieldVar("c"),
												right = Char('e')
											}
										},
										body = Return(Bool(false)),
										else = Some (
											If {
												cond = Binary {
													op = And,
													left = Binary {
														op = Equal,
														left = LocalOrFieldVar("c"),
														right = Char('f')
													},
													right = Binary {
														op = Equal,
														left = LocalOrFieldVar("c"),
														right = Char('g')
													}
												},
												body = Return(Bool(true)),
												else = Some (
													Block ([
														Return(Bool(false))
													])
												)
											}
										)
									}
								])
							)
						}
					)
				}
			])
		}
	]
}
