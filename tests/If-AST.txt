Class {
	name = "If",
	fields = [],
	methods = [
		Method {
			retType = boolean,
			name = "f",
			params = [
				VarDecl {
					type = char,
					name = "c",
					val = None
				}
			],
			body = Block([
				If {
					cond = Binary {
						op = Equal,
						left = LocalOrFieldVar("c"),
						right = Char('a')
					},
					body = Return(Bool(true)),
					else = None
				},
				Return(Bool(false))
			])
		}
	]
}
